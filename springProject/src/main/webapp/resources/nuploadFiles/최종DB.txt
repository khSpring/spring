-- order info view 

DROP VIEW ORDER_INFO;
-- 번호 회원아이디 결제방법 결제유무 배송단계 총합계 취소여부

CREATE VIEW ORDER_INFO AS
SELECT ROWNUM RNUM, O_NO, M_ID , PMNT_MTHD , PAID_YN ,
PRCS_STATUS , PMNT_PRICE  ,CANCEL_YN 
FROM(SELECT PI.O_NO , M.M_ID , PI.PMNT_MTHD , J.PAID_YN ,
J.PRCS_STATUS , PI.PMNT_PRICE ,J.CANCEL_YN 
FROM PAYMENT_INFO PI
JOIN MEMBER M ON (PI.M_NO = M.M_NO)
JOIN JUMUN J ON (PI.O_NO = J.O_NO)
ORDER BY PI.O_NO DESC);

-- 번호 회원아이디 결제방법 결제유무 배송단계 총합계 취소여부

COMMENT ON COLUMN ORDER_INFO.O_NO is '주문번호';
COMMENT ON COLUMN ORDER_INFO.M_ID is '회원아이디';
COMMENT ON COLUMN ORDER_INFO.PMNT_MTHD is '결제방법';
COMMENT ON COLUMN ORDER_INFO.PAID_YN is '결제유무';
COMMENT ON COLUMN ORDER_INFO.PRCS_STATUS is '배송단계';
COMMENT ON COLUMN ORDER_INFO.PMNT_PRICE is '총 합계';
